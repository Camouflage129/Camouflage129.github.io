```java
import java.util.Arrays;
import java.util.Scanner;
 
public class Solution {
    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        int T = scan.nextInt();
        for(int tc=1; tc<=T; tc++) {
            int N[] = new int[10];
            double sum = 0;
            for(int i=0; i<10; i++) 
                N[i] = scan.nextInt();
            Arrays.sort(N);
            for(int i=1; i<9; i++)
                sum += N[i];
            System.out.println("#"+tc+" "+Math.round(sum/8));
        }
        scan.close();
    }
}
```
