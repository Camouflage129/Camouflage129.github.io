~~~cpp
#include<iostream>
#include<string>
using namespace std;


int main(int argc, char** argv)
{
	string input;
	cin >> input;

	for (int i = 0; i < input.length(); i++) {
		if (97<=input.at(i) && input.at(i)<=122)
            printf("%c", input.at(i) - 32);
		else
			printf("%c", input.at(i));
	}

	return 0;
}
~~~
