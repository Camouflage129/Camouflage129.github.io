~~~cpp
#include<iostream>
#include <algorithm>
using namespace std;
 
int main(int argc, char** argv)
{
    int T;
    cin >> T;
    int *input = (int *)malloc(sizeof(int)*T);
 
    for (int test_case = 0; test_case < T; ++test_case)
    {
        cin >> input[test_case];
    }
    sort(input, input + T);
    printf("%d", input[T/2]);
 
    free(input);
    return 0;
}
~~~
