~~~cpp
#include<iostream>
#include<math.h>
using namespace std;
 
int process(int input, int count) {
    if (input / 10 == 0)
        return count;
    else
        process(input / 10, ++count);
}
 
int result(int input, int sum) {
    int num = process(input, 0);
    int temp;
    for (int i = 0;i <= num;i++) {
        temp = pow(10, num - i);
        sum += input / temp;
        input -= (input / temp) * temp;
    }
    return sum;
}
 
int main(int argc, char** argv)
{
    int T;
    cin >> T;
 
    printf("%d", result(T, 0));
    return 0;
}
~~~
