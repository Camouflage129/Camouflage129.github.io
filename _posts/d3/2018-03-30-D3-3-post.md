import java.util.Arrays;
import java.util.Scanner;
 
public class Solution {
 
    public static int getMax(int[][] arr) {
        int[] rSum = new int[100];
        int[] cSum = new int[100];
        int[] d1Sum = new int[100];
        int[] d2Sum = new int[100];
        int[] max = new int[4];
         
        // 행 합
        for (int i = 0; i < 100; i++) {
            for (int j = 0; j < 100; j++) {
                rSum[i] += arr[i][j];
            }
        }
        // 열 합
        for (int i = 0; i < 100; i++) {
            for (int j = 0; j < 100; j++) {
                cSum[i] += arr[j][i];
            }
        }
        // 대각선 합
        for (int i = 0; i < 100; i++) {
            d1Sum[i] += arr[i][i];
            d2Sum[i] += arr[i][99-i];
        }
        Arrays.sort(rSum);
        Arrays.sort(cSum);
        Arrays.sort(d1Sum);
        Arrays.sort(d2Sum);
         
        max[0] = rSum[99];
        max[1] = cSum[99];
        max[2] = d1Sum[99];
        max[3] = d2Sum[99];
         
        Arrays.sort(max);
        return max[3];
    }
 
    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
 
        for (int tc = 1; tc <= 10; tc++) {
            scan.nextInt();
            int[][] arr = new int[100][100];
            for (int i = 0; i < 100; i++) {
                for (int j = 0; j < 100; j++) {
                    arr[i][j] = scan.nextInt();
                }
            }
            System.out.println("#" + tc + " " + getMax(arr));
        }
        scan.close();
    }
}
