~~~cpp
#include<iostream>
 
using namespace std;
 
int main(int argc, char** argv)
{
    int test_case, T;
    int input;
    int sum = 0;
    cin >> T;
 
    for (test_case = 1; test_case <= T; ++test_case)
    {
        sum = 0;
        for (int i = 0; i < 10; i++) {
            cin >> input;
            if (input % 2 == 1)
                sum += input;
        }
        printf("#%d %d\n",test_case,sum);
    }
 
    return 0;
}
~~~
