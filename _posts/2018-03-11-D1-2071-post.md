~~~cpp
#include<iostream>
#include<math.h>
 
using namespace std;
 
int main(int argc, char** argv)
{
    int test_case, T;
    int input;
    int sum = 0;
    cin >> T;
 
    for (test_case = 1; test_case <= T; ++test_case)
    {
        sum = 0;
        for (int i = 0; i < 10; i++) {
            cin >> input;
            sum += input;
        }
        printf("#%d %.0lf\n",test_case,round((double)sum/10.0));
    }
 
    return 0;
}
~~~
