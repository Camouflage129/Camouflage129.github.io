```cpp
#include<iostream>
 
using namespace std;
 
char process(int input[]) {
    if (input[0] > input[1])
        return '>';
    else if (input[0] < input[1])
        return '<';
    else
        return '=';
}
 
int main(int argc, char** argv)
{
    int test_case, T;
    int input[2];
    cin >> T;
 
    for (test_case = 1; test_case <= T; ++test_case)
    {
        for (int i = 0; i < 2; i++) {
            cin >> input[i];
        }
        printf("#%d %c\n",test_case,process(input));
    }
 
    return 0;
}
```
