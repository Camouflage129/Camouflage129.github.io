```cpp
#include<iostream>
 
using namespace std;
 
int main(int argc, char** argv)
{
    int T;
    int input, max;
    cin >> T;
 
    for (int test_case = 1; test_case <= T; ++test_case)
    {
        max = 0;
        for (int i = 0; i < 10; i++) {
            cin >> input;
            if (input > max)
                max = input;
        }
        printf("#%d %d\n",test_case,max);
    }
 
    return 0;
}
```
